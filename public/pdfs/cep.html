Pure SQL-Based CEP

-- Detect fraud: multiple locations + high value
CREATE MATERIALIZED VIEW fraud_detection AS
SELECT userId,
  count(DISTINCT location) as locations,
  sum(amount) as total
FROM hop(transactions, 5s, 10m)
GROUP BY userId
HAVING locations >= 2 AND total > 1000
